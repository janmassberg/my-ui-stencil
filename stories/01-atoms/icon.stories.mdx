import { Canvas, Meta, Story, ArgsTable } from "@storybook/addon-docs";
import { html } from "lit-html";
import { ifDefined } from "lit-html/directives/if-defined";
import { IconName } from "../../src/components/icon/types";

export const allIconNames = Object.values(IconName);

export const storyWrapperRTL = Story => {
  return html`<div class="story-wrapper" dir="rtl">${Story()}</div>`;
};

export const Template = (args = {}) => {
  const { name, size, flipRtl } = args;
  return html`<ui-icon
    name=${ifDefined(name)}
    size=${ifDefined(size)}
    ?flip-rtl=${ifDefined(flipRtl)}
  ></ui-icon>`;
};

export const TemplateAll = (args = {}) => {
  return html`${allIconNames.map(name => Template({ ...args, name }))}`;
};

<Meta
  title="Atoms/Icon"
  component={`ui-icon`}
  argTypes={{
    name: {
      control: {
        type: "select",
        options: ["caret-down", "caret-left", "caret-right", "caret-up"],
      },
    },
    size: {
      control: {
        type: "radio",
        options: ["s", "m", "l"],
      },
    },
  }}
/>

# Icon

The icon component renders a SVGElement in one of three sizes.
Icons should have a viewBox of 24&times;24 pixels.

## Available Icons

<Canvas>
  <Story
    name="Available Icons"
    args={{
      size: "l",
    }}
  >
    {TemplateAll.bind({})}
  </Story>
</Canvas>

## Flip RTL

Set `flip-rtl` to flip the icon horizontally for right-to-left languages.

<Canvas>
  <Story
    name="Flip RTL"
    decorators={[storyWrapperRTL]}
    args={{
      size: "l",
      name: "caret-right",
      flipRtl: true,
    }}
  >
    {Template.bind({})}
  </Story>
</Canvas>
