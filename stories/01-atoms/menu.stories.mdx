import { Canvas, Meta, Story, ArgsTable } from "@storybook/addon-docs";
import { html } from "lit-html";
import { ifDefined } from "lit-html/directives/if-defined";

export const dummyMenuItems = [
  { name: "item--1", label: "Menu Item 1" },
  { name: "item--2", label: "Menu Item 2" },
  { name: "item--3", label: "Menu Item 3" },
  { name: "item--4", label: "Menu Item 4" },
];

export const Template = (args = {}) => {
  const { name, kind, current, items, _children = "" } = args;
  return html`<ui-menu
    name=${ifDefined(name)}
    kind=${ifDefined(kind)}
    current=${ifDefined(current)}
    .items=${ifDefined(items)}
    >${html([_children])}</ui-menu
  >`;
};

<Meta
  title="Atoms/Menu"
  component={`ui-menu`}
  argTypes={{
    name: {
      control: "text",
    },
    kind: {
      control: {
        type: "radio",
        options: ["primary", "secondary"],
      },
    },
    current: {
      control: "text",
    },
    _children: {
      control: "text",
    },
  }}
/>
